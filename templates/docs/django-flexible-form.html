Instalation

Instalation is very simple over pip. 
# pip install django-flexible-form
add to INSTALLED_APPS in settings.py 
INSTALLED_APPS = (
    # ...
    "flexible_form",
    # ...
)

Usage

Replace help texts

myapp/models.py

from django.db import models

class MyModel(models.Model):
    a = models.IntegerField(help_text="value of a")
    b = models.IntegerField()
    c = models.IntegerField()

myapp/forms.py

from django import forms
from flexible_form import FlexibleForm

from models import MyModel


class MyForm(forms.ModelForm, FlexibleForm):
    help_texts = {
        "a": "Insert value of a",
        "b": "Insert value of b",        
    }

    def __init__(self, *args, **kwargs):
        super(MyForm, self).__init__(*args, **kwargs)
        self.replace_help_texts()
