remoteforms documentation

Library for AJAX access to forms

Instalation

Instalation is very simple over pip. 
# pip install remoteforms

Djnago functions

from remoteforms import form_baseview, done_baseview

http_response, object or None = form_baseview(request, form_class, template, save)

http_response = form_baseview(request, template)

JavaScript functions

remoteForm(formDiv, formID, submitID, formURL, doneURL)

Usage django

settings.py

INSTALLED_APPS = (
    ...
    "remoteforms",
    ...
)

views.py

from remoteforms import form_baseview, done_baseview
from forms import MyForm

def my_form_view(request, template="my_form.html"):
    return form_baseview(request, MyForm, template, True)[0]


def my_done_view(request, template="my_form_done.html"):
    return form_baseview(request, template)

Usage JavaScript

<!doctype html>
<html lang="en">
<head>
  <meta charset="utf-8">
  <title></title>
  <script type="text/javascript" src="src/jquery-1.9.1.min.js"></script>
  <script type="text/javascript" src="src/remote-forms.js"></script>
  <script type="text/javascript">
    $(document).ready(function (){
        remoteForm("#form", "form", "#formsubmit", "http://localhost:8000/form/", "http://localhost:8000/done/");
    })
  </script>
</head>
<body>
    <div id="form"></div><button id="formsubmit">submit</button>
</body>
</html>
